## Schema (Prisma)

```prisma
// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ==================== 用户 ====================
model User {
  id            String    @id @default(cuid())
  email         String    @unique
  name          String?
  avatar        String?   // URL 或 base64
  credits       Int       @default(100)
  role          UserRole  @default(USER)
  
  // OAuth
  accounts      Account[]
  sessions      Session[]
  
  // 用户行为
  ratings       Rating[]
  toolUsages    ToolUsage[]
  favorites     Favorite[]
  achievements  UserAchievement[]
  leaderboard   LeaderboardEntry[]
  wishlistItems WishlistItem[]
  
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
  
  @@map("users")
}

enum UserRole {
  USER
  ADMIN
  MODERATOR
}

// ==================== NextAuth ====================
model Account {
  id                String  @id @default(cuid())
  userId            String
  type              String
  provider          String
  providerAccountId String
  refresh_token     String? @db.Text
  access_token      String? @db.Text
  expires_at        Int?
  token_type        String?
  scope             String?
  id_token          String? @db.Text
  session_state     String?

  user User @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@unique([provider, providerAccountId])
  @@map("accounts")
}

model Session {
  id           String   @id @default(cuid())
  sessionToken String   @unique
  userId       String
  expires      DateTime
  user         User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  @@map("sessions")
}

// ==================== 工具 ====================
model Tool {
  id          String       @id @default(cuid())
  slug        String       @unique  // URL 路径，如 "json-formatter"
  name        String       // 显示名称
  nameEn      String?      // 英文名称
  category    ToolCategory
  icon        String       // emoji 或 SVG 路径
  description String       // 一句话描述
  creditCost  Int          @default(0)  // 0 = 免费
  tags        String[]     // 标签数组
  url         String       // 页面路径，如 "/tools/json"
  order       Int          @default(0)  // 排序权重
  
  // 统计数据（由触发器或定时任务更新）
  avgRating   Float?       @default(0)
  totalRatings Int         @default(0)
  totalUses   Int          @default(0)
  
  ratings     Rating[]
  usages      ToolUsage[]
  favorites   Favorite[]
  
  createdAt   DateTime     @default(now())
  updatedAt   DateTime     @updatedAt
  
  @@map("tools")
}

enum ToolCategory {
  DEV          // 开发工具
  UTILITY      // 实用工具
  AI_WRITING   // AI 写作工具
  GAME         // 小游戏
}

// ==================== 评分 ====================
model Rating {
  id        String   @id @default(cuid())
  userId    String
  toolId    String
  score     Int      // 1-5 星
  review    String?  @db.Text  // 评论内容
  
  user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  tool      Tool     @relation(fields: [toolId], references: [id], onDelete: Cascade)
  
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  
  @@unique([userId, toolId])  // 一个用户只能评一次
  @@map("ratings")
}

// ==================== 工具使用记录 ====================
model ToolUsage {
  id       String   @id @default(cuid())
  userId   String
  toolId   String
  usedAt   DateTime @default(now())
  duration Int?     // 使用时长（秒）
  
  user     User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  tool     Tool     @relation(fields: [toolId], references: [id], onDelete: Cascade)
  
  @@map("tool_usages")
}

// ==================== 收藏 ====================
model Favorite {
  id        String   @id @default(cuid())
  userId    String
  toolId    String
  createdAt DateTime @default(now())
  
  user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  tool      Tool     @relation(fields: [toolId], references: [id], onDelete: Cascade)
  
  @@unique([userId, toolId])
  @@map("favorites")
}

// ==================== 排行榜 ====================
model LeaderboardEntry {
  id        String         @id @default(cuid())
  userId    String
  score     Int
  period    LeaderboardPeriod
  rank      Int?
  
  user      User           @relation(fields: [userId], references: [id], onDelete: Cascade)
  
  createdAt DateTime       @default(now())
  updatedAt DateTime       @updatedAt
  
  @@unique([userId, period])
  @@map("leaderboard")
}

enum LeaderboardPeriod {
  DAILY
  WEEKLY
  MONTHLY
  ALL_TIME
}

// ==================== 成就 ====================
model Achievement {
  id          String   @id @default(cuid())
  slug        String   @unique
  name        String
  nameEn      String?
  description String
  icon        String   // emoji 或图标路径
  condition   String   // 解锁条件描述（如 "使用10个不同工具"）
  points      Int      @default(10)  // 成就奖励积分
  
  users       UserAchievement[]
  
  @@map("achievements")
}

model UserAchievement {
  id            String   @id @default(cuid())
  userId        String
  achievementId String
  unlockedAt    DateTime @default(now())
  
  user          User         @relation(fields: [userId], references: [id], onDelete: Cascade)
  achievement   Achievement  @relation(fields: [achievementId], references: [id], onDelete: Cascade)
  
  @@unique([userId, achievementId])
  @@map("user_achievements")
}

// ==================== 许愿单 ====================
model WishlistItem {
  id        String   @id @default(cuid())
  userId    String
  title     String
  description String?
  votes     Int      @default(0)
  status    WishlistStatus @default(PENDING)
  
  user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
  
  @@map("wishlist_items")
}

enum WishlistStatus {
  PENDING
  APPROVED
  IN_PROGRESS
  DONE
  REJECTED
}
```
